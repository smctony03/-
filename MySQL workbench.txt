-- 1. 사용할 데이터베이스 명시
USE food_db;

-- 2. 기존 테이블 삭제 (구조를 새로 잡기 위함)
DROP TABLE IF EXISTS food;

-- 3. 5가지 카테고리가 포함된 테이블 생성
CREATE TABLE food (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(50) NOT NULL,
    img_url TEXT,                 -- 긴 URL을 위해 TEXT 사용
    country VARCHAR(20),          -- 나라: 한식, 중식 등
    taste VARCHAR(20),            -- 맛: 자극적인 맛, 담백한 맛
    main_ingredient VARCHAR(20),  -- 주재료: 육고기, 채소, 빵 등
    temperature VARCHAR(20),      -- 온도: 따뜻한 것, 차가운 것
    cooking_type VARCHAR(20)      -- 유형: 구운 것, 볶은 것, 국물 등
);
